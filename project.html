Display the city name from given API: https://jsonplaceholder.typicode.com/users var xhr = new XMLHttpRequest(); xhr.open("GET", "https://jsonplaceholder.typicode.com/users"); xhr.onload = function (){ if(xhr.status >= 200 && xhr.status < 300){ let data = JSON.parse(this.response); console.log(data) for(let i = 0; i < data.length; i++){ let user = data[i]; console.log(user.address.city); } } }; xhr.send();
